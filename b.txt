 * Converts yyyy-MM-dd â†’ MM/dd/yyyy
 * Safe for ASP.NET WebForms inputs
 */
export function formatDateToUs(date: string | null | undefined): string {
  if (!date) return "";

  // Expecting yyyy-MM-dd
  const parts = date.split("-");
  if (parts.length !== 3) return date;

  const [year, month, day] = parts;
  return `${month}/${day}/${year}`;
}
