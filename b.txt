function decodeJsEncodedString(s) {
    if (s == null) return s;

    return s
        .replace(/\\u003c/g, "<")
        .replace(/\\u003e/g, ">")
        .replace(/\\u0026/g, "&")
        .replace(/\\u0027/g, "'")
        .replace(/\\u0022/g, "\"")
        .replace(/\\r\\n|\\n|\\r/g, "\n")
        .replace(/\\\\/g, "\\");
}
