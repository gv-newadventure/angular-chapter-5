```ts
export function bindClick(
  selector: string,
  handler: () => void
): void {
  const el = document.querySelector(selector) as HTMLElement | null;
  if (!el) return;

  el.onclick = handler;
}
```

## ‚ö†Ô∏è Option B: Pure DOM close (NO jQuery UI close)

Only use this **if you plan to remove jQuery UI soon**.

```ts
export function wireVendorNoMatchOkButton(): void {
  const btn = document.getElementById("btnAiVendorNoMatchOk");
  const dialog = document.getElementById("divAiVendorNoMatch");

  if (!btn || !dialog) return;

  btn.addEventListener("click", () => {
    dialog.style.display = "none";
  });
}
```

‚ùå Does NOT call `dialog("close")`
‚ùå Skips jQuery UI lifecycle
‚úî Useful for future refactor
