-- After both branches have run:
INSERT dbo.tbl_WorkflowUsers
(
    WorkflowKey,
    Status,
    OpenDate,
    StatusDate,
    UserKey,
    Step,
    Type,
    ApprovalAmount
)
SELECT
    IM.NewWorkflowKey,
    0,
    SYSDATETIME(),
    SYSDATETIME(),
    @WorkFlowUserKey,
    1,
    1,
    NULL
FROM @EFTWorkflowMapping IM
WHERE NOT EXISTS (
    SELECT 1
    FROM dbo.tbl_WorkflowUsers WU
    WHERE WU.WorkflowKey = IM.NewWorkflowKey
);
