/**
 * Adds AI highlight styling to an input or element
 */
export function highlightAiField(el: HTMLElement): void {
  el.classList.add("ai-highlight");
}

/**
 * Safe helper to get an input element by id
 */
export function getInputElement(id: string): HTMLInputElement | null {
  return document.getElementById(id) as HTMLInputElement | null;
}

/**
 * Safe helper to get any element by id
 */
export function getElement(id: string): HTMLElement | null {
  return document.getElementById(id);
}
